358 BigTree BigTree::getChildWithProperty(const Identifier &propertyName, const var &propertyValue, bool recursive)
359 {
360  //checking root node first
361  if (hasProperty(propertyName))
362  if (getProperty(propertyName) == propertyValue)
363  return *this;
364 
365  BigTree bTree(ValueTree::getChildWithProperty(propertyName, propertyValue));
366  if (bTree.isValid()) {
367  return bTree;
368  } else if (!recursive) {
369  return (BigTree());
370  }
371 
372  for (int i = getNumChildren(); --i >= 0;) {
373  BigTree cTree(getChild(i).getChildWithProperty(propertyName, propertyValue, recursive));
374  if (cTree.isValid()) {
375  return cTree;
376  }
377  }
378  return (BigTree());
379 }
