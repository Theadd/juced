44  void valueTreePropertyChanged (ValueTree &treeWhosePropertyHasChanged, const Identifier &property)
45  {
46  //Deal with changes on this ValueTree only, not their childs.
47  if (treeWhosePropertyHasChanged == *this) {
48 
49  jassert (this->hasProperty(Attributes::objectType)); //Object requires: setProperty(Ids::objectType, "object named type")
50 
51  //since Windows don't inherit from Component class we deal with it separately
52  if (this->getProperty(Attributes::objectType) == Modules::Window) {
53  juced_Window *obj = (juced_Window *)this->getProperty(Attributes::object).getDynamicObject();
54  if (property == Attributes::height) {
55  int height = treeWhosePropertyHasChanged.getProperty(property);
56  obj->setSize(obj->getWidth(), height);
57  } else if (property == Attributes::width) {
58  int width = treeWhosePropertyHasChanged.getProperty(property);
59  obj->setSize(width, obj->getHeight());
60  } else if (property == Attributes::name) {
61  obj->setName(treeWhosePropertyHasChanged.getProperty(property).toString());
62  } else if (property == Attributes::backgroundColour) {
63  obj->setName(treeWhosePropertyHasChanged.getProperty(property).toString());
64  Colour newColour = Colour::fromString(treeWhosePropertyHasChanged.getProperty(property).toString());
65  obj->setColour(DocumentWindow::backgroundColourId, newColour);
66  }
67  } else {
68  //deal with other type of components since its not a window
69  bool propertyChanged = true;
70  if (this->getProperty(Attributes::objectType) == Modules::Label) {
71  //deal with Label specific properties
72  juced_Label *cObject = (juced_Label *)this->getProperty(Attributes::object).getDynamicObject();
73  if (property == Attributes::text) {
74  cObject->setText(treeWhosePropertyHasChanged.getProperty(property).toString(), false);
75  } else if (property == Attributes::backgroundColour) {
76  Colour newColour = Colour::fromString(treeWhosePropertyHasChanged.getProperty(property).toString());
77  cObject->setColour(Label::backgroundColourId, newColour);
78  } else if (property == Attributes::textColour) {
79  Colour newColour = Colour::fromString(treeWhosePropertyHasChanged.getProperty(property).toString());
80  cObject->setColour(Label::textColourId, newColour);
81  } else {
82  propertyChanged = false;
83  }
84  } else if (this->getProperty(Attributes::objectType) == Modules::TextButton) {
85  //deal with TextButton specific properties
86  juced_TextButton *cObject = (juced_TextButton *)this->getProperty(Attributes::object).getDynamicObject();
87  if (property == Attributes::buttonText) {
88  cObject->setButtonText(treeWhosePropertyHasChanged.getProperty(property).toString());
89  } else if (property == Attributes::buttonColour) {
90  Colour newColour = Colour::fromString(treeWhosePropertyHasChanged.getProperty(property).toString());
91  cObject->setColour(TextButton::buttonColourId, newColour);
92  } else if (property == Attributes::buttonOnColour) {
93  Colour newColour = Colour::fromString(treeWhosePropertyHasChanged.getProperty(property).toString());
94  cObject->setColour(TextButton::buttonOnColourId, newColour);
95  } else {
96  propertyChanged = false;
97  }
98  } else if (this->getProperty(Attributes::objectType) == Modules::Component) {
99  //deal with Component specific properties (obviously is empty)
100  propertyChanged = false;
101  }
102  if (!propertyChanged) {
103  //deal with common component properties
104  Component *obj = dynamic_cast<Component *> (this->getProperty(Attributes::object).getDynamicObject());
105  if (property == Attributes::x) {
106  int value = treeWhosePropertyHasChanged.getProperty(property);
107  obj->setBounds(value, obj->getY(), obj->getWidth(), obj->getHeight());
108  } else if (property == Attributes::y) {
109  int value = treeWhosePropertyHasChanged.getProperty(property);
110  obj->setBounds(obj->getX(), value, obj->getWidth(), obj->getHeight());
111  } else if (property == Attributes::width) {
112  int value = treeWhosePropertyHasChanged.getProperty(property);
113  obj->setBounds(obj->getX(), obj->getY(), value, obj->getHeight());
114  } else if (property == Attributes::height) {
115  int value = treeWhosePropertyHasChanged.getProperty(property);
116  obj->setBounds(obj->getX(), obj->getY(), obj->getWidth(), value);
117  } else if (property == Attributes::name) {
118  obj->setName(treeWhosePropertyHasChanged.getProperty(property).toString());
119  }
120  }
121  }
122  //Set current transaction description of UndoManager and start a new one
123  Attribute *t = Constructor::getInstance()->getAttributeOf(property);
124  Constructor::getInstance()->getUndoManager()->setCurrentTransactionName("Set " + t->display + " on " + getProperty(Attributes::name).toString());
125  Constructor::getInstance()->getUndoManager()->beginNewTransaction();
126  }
127  }

155  BigTree& getChildWithProperty(const Identifier &propertyName, const var &propertyValue, bool recursive)
156  {
157  //checking root node first
158  if (hasProperty(propertyName))
159  if (getProperty(propertyName) == propertyValue)
160  return *this;
161 
162  BigTree bTree(ValueTree::getChildWithProperty(propertyName, propertyValue));
163  if (bTree.isValid()) {
164  return bTree;
165  } else if (!recursive) {
166  return (BigTree());
167  }
168 
169  for (int i = getNumChildren(); --i >= 0;) {
170  BigTree cTree(getChild(i).getChildWithProperty(propertyName, propertyValue, recursive));
171  if (cTree.isValid()) {
172  return cTree;
173  }
174  }
175  return (BigTree());
176  }
