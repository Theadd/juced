75 void CodeGenerator::setSpecificProperty(Identifier prop, String appliedTo)
76 {
77  Attribute *attrib = Constructor::getInstance()->getAttributeOf(prop);
78  String value = replaceKeywords("%" + attrib->name.toString() + "%");
79  /* prop = Property identifier
80  value = Property value as string
81  attrib = pointer to Attribute struct containing attribute.xml fields for specified property
82  */
83  if (prop == Attributes::itemList) {
84  StringArray itemList;
85  itemList.addLines(value);
86  for (int i = 0; i < itemList.size(); ++i) {
87  _definition += appliedTo + "addItem(\"" + itemList[i] + "\", " + String(i + 1) + ");\n";
88  }
89  } else if (prop == Attributes::lookAndFeel) {
90  if (value != "Inherit") {
91  _definition += appliedTo + "setLookAndFeel(" + _tree->getProperty(Attributes::lookAndFeel).toString() + "::getInstance());\n";
92  }
93  } else if (prop == Attributes::castMouseEvents) {
94  bool castMouseEvents = _tree->getProperty(Attributes::castMouseEvents);
95  if (castMouseEvents) {
96  _definition += appliedTo + "addMouseListener(this, false);\n";
97  }
98  } else if (prop == Attributes::showEditor) {
99  bool showEditor = _tree->getProperty(Attributes::showEditor);
100  if (showEditor) {
101  _definition += appliedTo + "showEditor();\n";
102  }
103  } else if (prop == Attributes::normalImage) {
104  String normalImage = (File::isAbsolutePath(value)) ? "ImageFileFormat::loadFrom(File::getCurrentWorkingDirectory().getChildFile(\"../../Source/BinaryData/" + File(value).getFileName() + "\"))" : "Image()";
105  String overImage = (File::isAbsolutePath(_tree->getProperty(Attributes::overImage))) ? "ImageFileFormat::loadFrom(File::getCurrentWorkingDirectory().getChildFile(\"../../Source/BinaryData/" + File(_tree->getProperty(Attributes::overImage)).getFileName() + "\"))" : "Image()";
106  String downImage = (File::isAbsolutePath(_tree->getProperty(Attributes::downImage))) ? "ImageFileFormat::loadFrom(File::getCurrentWorkingDirectory().getChildFile(\"../../Source/BinaryData/" + File(_tree->getProperty(Attributes::downImage)).getFileName() + "\"))" : "Image()";
107  _definition += appliedTo + "setImages (false, true, true, " + normalImage + ", 0.6000f, Colour((uint8) 0, (uint8) 0, (uint8) 0, 0.0f), " + overImage + ", 1.0000f, Colour((uint8) 0, (uint8) 0, (uint8) 0, 0.0f), " + downImage + ", 1.0000f, Colour((uint8) 150, (uint8) 150, (uint8) 150, 0.3f));\n";

113 void CodeGenerator::refresh()
114 {
115  Constructor::log("CG002 - CodeGenerator refresh call");
116  _childGenerators.clear();
117  _code = "";
118 
119  getKeywords();
120  Constructor *constructor = Constructor::getInstance();
121 
122  bool declareExtended = false;
123  if (_tree->hasProperty(Attributes::declareExtended)) {
124  declareExtended = _tree->getProperty(Attributes::declareExtended);
125  }
126  String appliedTo = "";
127  if (declareExtended) {
128  //get respective template and replace its keywords
129  _code = getTemplate();
130  _code = replaceKeywords(_code);
131  } else {
132  _declaration = _tree->getProperty(Attributes::className).toString() + " " + _tree->getProperty(Attributes::varName).toString() + ";\n";
133  appliedTo = _tree->getProperty(Attributes::varName).toString() + ".";
134  }
135  //get definition code adding attributes if its syntax is specified
136  for (int i = _tree->getNumProperties(); --i >= 0;) {
137  Attribute *attrib = constructor->getAttributeOf(_tree->getPropertyName(i));
138  if (attrib != nullptr && !attrib->syntax.isEmpty()) {
139  if (attrib->syntax == "#") {
140  setSpecificProperty(attrib->name, appliedTo);
141  } else {
142  _definition += appliedTo + replaceKeywords(attrib->syntax) + ";\n";
143  }
144  }
145  }
146 
147  if (!declareExtended) {
148  //if parent component is a window, this component must be owned by the window
149  if (_parentCodeGenerator->isContentOwner()) {
150  //if parent window is not an extended DocumentWindow class, set to modify its corresponding object
151  if (!_parentCodeGenerator->isDeclaredAsExtended()) {
152  _definition += _parentCodeGenerator->getVarName() + ".";
153  }
154  //set this component as owned by the parent window
155  _definition += "setContentOwned (&" + _tree->getProperty(Attributes::varName).toString() + ", false);\n";
156  } else {
157  //add and make visible this component into the parent component
158  if (!_parentCodeGenerator->isDeclaredAsExtended()) {
159  _definition += _parentCodeGenerator->getVarName() + ".";
160  }
161  _definition += "addAndMakeVisible(&" + _tree->getProperty(Attributes::varName).toString() + ");\n";
162  }
163  }
164 
165  //refresh childs
166  for (int i = _tree->getNumChildren(); --i >= 0;) {
167  BigTree cTree(_tree->getChild(i));
168  if (cTree.isValid()) {
169  _childGenerators.add(new CodeGenerator(cTree, this));
170  }
171  }
172
