203 void Constructor::log(String msg)
204 {
205  File logFile(File::addTrailingSeparator(File::getCurrentWorkingDirectory().getFullPathName()) + "designer.log");
206  //Create log file if it does not exists
207  if (!logFile.exists())
208  logFile.create();
209  //Apply format
210  String::CharPointerType pointer(msg.getCharPointer());
211  for (; !pointer.isDigit(); pointer++);
212  int logType = (int)(pointer[0]) - 48;
213  switch (logType) {
214  case 0:
215  msg = Time::getCurrentTime().formatted("%M:%S") + ": " + msg;
216  break;
217  case 1:
218  msg = "\t[INFO] " + Time::getCurrentTime().formatted("%M:%S") + ": " + msg;
219  break;
220  case 2:
221  msg = "\t[WARNING] " + Time::getCurrentTime().formatted("%M:%S") + ": " + msg;
222  break;
223  case 3:
224  msg = "\t[ERROR] " + Time::getCurrentTime().formatted("%M:%S") + ": " + msg;
225  break;
226  }
227  //Append msg to log file
228  logFile.appendText(msg + "\n");
229 }

263 Component* Constructor::createComponent(String selectedToolName, String parentComponentID, Rectangle<int> bounds)
264 {
265  PlaceableComponent *newComponent = new PlaceableComponent(selectedToolName, parentComponentID, bounds);
266  _placeableComponents.add(newComponent);
267  getUndoManager()->perform(newComponent, "Create new " + selectedToolName);
268  //getUndoManager()->setCurrentTransactionName("Create new " + selectedToolName);
269  if (selectedToolName == "juced_Window") {
270  bounds.setX(0);
271  bounds.setY(0);
272  bounds.setHeight(bounds.getHeight() - (dynamic_cast<DocumentWindow*> (newComponent->getComponent()))->getTitleBarHeight());
273  PlaceableComponent *childComponent = new PlaceableComponent("juced_MainComponent", newComponent->getComponent()->getComponentID(), bounds);
274  _placeableComponents.add(childComponent);
275  getUndoManager()->perform(childComponent);
276  (dynamic_cast<DocumentWindow*> (newComponent->getComponent()))->setContentOwned(childComponent->getComponent(), true);
277  }
278  getUndoManager()->beginNewTransaction();
279  return newComponent->getComponent();
280 }
