50 bool PlaceableComponent::perform ()
51 {
52  Constructor::log("PC001 - Perform init (" + _selectedToolName + ")");
53  bool isBeingCreated = false;
54  Constructor::log("PC101 - Find associated BigTree of the parent component");
55  //If user draw inside a component, let's find it's associated BigTree of the parent component
56  BigTree parentTree;
57  BigTree *bigTree = Constructor::getInstance()->getBigTreeRoot();
58  if (_parentComponentID.isNotEmpty() && bigTree != nullptr) {
59  BigTree bTree(bigTree->getChildWithProperty(Attributes::ID, _parentComponentID, true));
60  if (bTree.isValid()) parentTree = bTree;
61  }
62 
63  //If user is placing the component as a root component and there was already one don't create the component
64  if (!parentTree.isValid() && Constructor::getInstance()->getBigTreeRoot() != nullptr) {
65  Constructor::log("PC301 - BigTree root was already specified");
66  return false;
67  }
68 
69  //If component does not exist yet, create an instance of it
70  if (_dynamicObject == nullptr) {
71  Constructor::log("PC101 - Component does not exist yet, creating an instance of it");
72  isBeingCreated = true;
73  if ((_dynamicObject = createObjectFromToolName(&_selectedToolName)) == nullptr) {
74  //selectedToolName does not match any object type
75  return false;
76  }
77  }
78 
79  Constructor::log("PC101 - Get parent component if parent BigTree is valid, editor component otherwise");
80  //get parent component if parent BigTree is valid, editor component otherwise
81  Component *parentComponent;
82  if (parentTree.isValid()) {
83  parentComponent = dynamic_cast<Component *> (parentTree.getProperty(Attributes::object).getDynamicObject());
84  } else {
85  parentComponent = Constructor::getInstance()->getDesigner();
86  }
87 
88  Constructor::log("PC101 - Place component inside parent component and add its mouse listener");
89  //Place component inside parent component and add its mouse listener
90  parentComponent->addAndMakeVisible(dynamic_cast<Component *> (_dynamicObject.get()));
91  (dynamic_cast<Component *> (_dynamicObject.get()))->addMouseListener(Constructor::getInstance()->getDesigner(), false);
92  /*if (isBeingCreated) {
93  Constructor::log("PC101 - Set component bounds");
94  (dynamic_cast<Component *> (_dynamicObject))->setBounds(_bounds);
95  }*/
96 
97  Constructor::log("PC101 - Update component's ID if specified");
98  //update component's ID if specified
99  if (_componentID.isNotEmpty()) {
100  (dynamic_cast<Component *> (_dynamicObject.get()))->setComponentID(_componentID);
101  _dynamicObject->setProperty(Attributes::ID, _componentID);
102  } else _componentID = _dynamicObject->getProperty(Attributes::ID);
103 
104  Constructor::log("PC101 - Get component's BigTree or create a new one if it is invalid");
105  //get component's BigTree or create a new one if it is invalid
106  BigTree *objTree = new BigTree();
107  if (!_componentTree->isValid()) {
108  //current component's BigTree is invalid, create a new one
109  Constructor::log("PC101 - current component's BigTree is invalid, create a new one");
110  delete _componentTree;
111  Constructor::log("PC101 - deleted");
112  objTree = new BigTree(_dynamicObject, _dynamicObject->getProperty(Attributes::objectType));
113  Constructor::log("PC101 - created, valid: " + String(objTree->isValid()) + " @" + String((int)&_componentTree));
114  //DBG((static_cast<void *> (&_componentTree)));
115  *_componentTree = BigTree(*objTree);
116  Constructor::log("PC101 - copied");
117  } else {
118  Constructor::log("PC101 - else!! wtf");
119  //get current component's BigTree
120  *objTree = BigTree(*_componentTree);
121  //objTree->setDynamicObject(_dynamicObject);
122  jassert (objTree->isValid());
123  }
124  Constructor::log("PC101 - done getting component's bigtree");
125 
126  //Add component's BigTree as a child of parent's BigTree
127  if (parentTree.isValid()) {
128  Constructor::log("PC101 - Add component's BigTree as a child of parent's BigTree");
129  parentTree.addChild(*objTree, -1, 0);
130  } else {
131  //There was no parent BigTree so we assume this is going to be the main component container
132  Constructor::log("PC101 - Set BigTree root as this component's BigTree");
133  objTree->setProperty(Attributes::declareExtended, true, 0);
134  Constructor::getInstance()->setBigTreeRoot(objTree);
135  }
136 
137  //Set component bounds.
138  if (isBeingCreated) {
139  Constructor::log("PC101 - Set component bounds");
140  objTree->setProperty(Attributes::x, _bounds.getX(), 0);
141  objTree->setProperty(Attributes::y, _bounds.getY(), 0);
142  objTree->setProperty(Attributes::width, _bounds.getWidth(), 0);
143  objTree->setProperty(Attributes::height, _bounds.getHeight(), 0);
144  }
145 
146  //set common value properties
147  if (!objTree->hasProperty(Attributes::lookAndFeel))
148  objTree->setProperty(Attributes::lookAndFeel, String("Inherit"), 0);
149  if (!objTree->hasProperty(Attributes::castMouseEvents))
150  objTree->setProperty(Attributes::castMouseEvents, false, 0);
151 
152  //if parent component is a component owner (must own its child), specify it
153  if (parentTree.isValid() && parentTree.hasProperty(Attributes::contentOwner)) {
154  bool contentOwner = parentTree.getProperty(Attributes::contentOwner);
155  if (contentOwner) {
156  ResizableWindow *parentWindow = dynamic_cast<ResizableWindow *> (parentTree.getProperty(Attributes::object).getDynamicObject());
157  parentWindow->setContentOwned(dynamic_cast<Component *> (_dynamicObject.get()), false);
158  }
159  } else if (parentTree.isValid() && parentTree.hasProperty(Attributes::contentViewed)) {
160  bool contentViewed = parentTree.getProperty(Attributes::contentViewed);
161  if (contentViewed) {
162  Viewport *parentViewport = dynamic_cast<Viewport *> (parentTree.getProperty(Attributes::object).getDynamicObject());
163  parentViewport->setViewedComponent(dynamic_cast<Component *> (_dynamicObject.get()), false);
164  }
165  }
166 
167  Constructor::log("PC101 - Perform done");
168  return true;
