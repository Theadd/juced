116 void JUCE_Designer::selectComponent (Component *componentToSelect)
117 {
118  if (componentToSelect != this) {
119  selectionBox->setVisible(false);
120  Point<int> pos = componentToSelect->getScreenPosition() - this->getScreenPosition();
121  selectedComponentPositionDifference = pos - componentToSelect->getPosition();
122  selectionBox->setSelectionBounds(pos.getX(), pos.getY(), componentToSelect->getWidth(), componentToSelect->getHeight());

193 void JUCE_Designer::mouseUp (const MouseEvent& event)
194 {
195  String *selectedToolName = getSelectedToolName();
196  if (!event.mouseWasClicked()) {
197  if (selectedToolName->isNotEmpty()) {
198 
199  BigTree parentTree;
200  if (event.originalComponent != this) {
201  BigTree bTree(bigTree->getChildWithProperty(Attributes::ID, event.originalComponent->getComponentID(), true));
202  if (bTree.isValid()) parentTree = bTree;
203  }
204 
205  MouseEvent relativeEvent = event.getEventRelativeTo(event.originalComponent);
206 
207  if (selectedToolName->equalsIgnoreCase("juced_Window")) {
208  addWindow(event.originalComponent, relativeEvent.getMouseDownX(), relativeEvent.getMouseDownY(), relativeEvent.getDistanceFromDragStartX(), relativeEvent.getDistanceFromDragStartY());
209  } else {
210  //Create a component of the selected tool name unless it is placed outside a window
211  if (parentTree.isValid()) {
212  DynamicObject *dynamicObj;
213  if ((dynamicObj = createObjectFromToolName(selectedToolName)) != nullptr) {
214 
215  event.originalComponent->addAndMakeVisible(dynamic_cast<Component *> (dynamicObj));
216  (dynamic_cast<Component *> (dynamicObj))->addMouseListener(this, false);
217 
218  BigTree *objTree = new BigTree(dynamicObj, dynamicObj->getProperty(Attributes::objectType));
219 
220  parentTree.addChild(*objTree, -1, 0);
221 
222  objTree->setProperty(Attributes::x, relativeEvent.getMouseDownX(), 0);
223  objTree->setProperty(Attributes::y, relativeEvent.getMouseDownY(), 0);
224  objTree->setProperty(Attributes::width, relativeEvent.getDistanceFromDragStartX(), 0);
225  objTree->setProperty(Attributes::height, relativeEvent.getDistanceFromDragStartY(), 0);
