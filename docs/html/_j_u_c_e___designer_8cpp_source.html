238 bool JUCE_Designer::keyPressed (const KeyPress& key)
239 {
240  mousePositionLabel.setText(String(key.getKeyCode()), 0);
241  if (key.getKeyCode() == 90 && key.getModifiers().isCtrlDown()) {
242  Constructor::getInstance()->getSelectionBox()->setVisible(false);

244  Constructor::getInstance()->getUndoManager()->undo();

249  Constructor::getInstance()->getUndoManager()->redo();

252  int rand = 0;

254  Constructor::getInstance()->setGridSize(rand);
255  grid.repaint();

257  } else if (key.getKeyCode() == 83 && key.getModifiers().isCtrlDown()) {
258  this->writeXmlToFile("save.xml");
259  } else if (key.getKeyCode() == key.leftKey) {
260  if (Constructor::getInstance()->getSelectedComponent() != nullptr) {
261  BigTree valueTree(Constructor::getInstance()->getBigTreeRoot().getChildWithProperty(Attributes::ID, Constructor::getInstance()->getSelectedComponent()->getComponentID(), true));
262  int currentPosX = valueTree.getProperty(Attributes::x);

264  }
265  } else if (key.getKeyCode() == key.rightKey) {
266  if (Constructor::getInstance()->getSelectedComponent() != nullptr) {

271  } else if (key.getKeyCode() == key.upKey) {
272  if (Constructor::getInstance()->getSelectedComponent() != nullptr) {
273  BigTree valueTree(Constructor::getInstance()->getBigTreeRoot().getChildWithProperty(Attributes::ID, Constructor::getInstance()->getSelectedComponent()->getComponentID(), true));
274  int currentPosY = valueTree.getProperty(Attributes::y);
275  valueTree.setProperty(Attributes::y, currentPosY - 1, Constructor::getInstance()->getUndoManager());

278  if (Constructor::getInstance()->getSelectedComponent() != nullptr) {
279  BigTree valueTree(Constructor::getInstance()->getBigTreeRoot().getChildWithProperty(Attributes::ID, Constructor::getInstance()->getSelectedComponent()->getComponentID(), true));

317 void JUCE_Designer::Grid::paint (Graphics& g)
318 {
319  float gridSize = (float) Constructor::getInstance()->getGridSize();
320  int doubleLineInterval = 85 - (85 % (int) gridSize);
321  g.fillAll (Colours::transparentWhite);
322  g.setColour (Colours::white);
323  for (float x = gridSize; x < (float) getWidth(); x += gridSize) {
324  g.setOpacity((((int) x) % doubleLineInterval == 0) ? 0.15f : 0.06f);
325  g.drawVerticalLine((int)x, 0.0f, (float) getHeight());
326  g.drawHorizontalLine((int)x, 0.0f, (float) getWidth());
327  if (((int) x) % doubleLineInterval == 0) {
328  g.setOpacity(0.35f);
329  for (float i = (float) (int)(getHeight() / doubleLineInterval + 1); --i > 0;) {
330  g.drawVerticalLine((int)x, ((float) doubleLineInterval) * i - 3.0f, ((float) doubleLineInterval) * i + 4.0f);
331  g.drawHorizontalLine(doubleLineInterval * ((int) i), x - 3.0f, x + 4.0f);
332  }
333  }
334  }
335 }
